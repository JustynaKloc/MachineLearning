def Poli_reg_w(X,Y):

    x = X[:, np.newaxis]
    y = Y[:, np.newaxis]
    r=[]
    
    for i in range(2,22):
        polynomial_features= PolynomialFeatures(degree=i)
        x_poly = polynomial_features.fit_transform(x)

        model = LinearRegression()
        model.fit(x_poly, y)
        y_poly_pred = model.predict(x_poly)
    
        rmse = np.sqrt(mean_squared_error(y,y_poly_pred))
        r2 = r2_score(y,y_poly_pred)
        r.append((r2,i))
    #print(rmse)
    #print(r2)
    #print(r)
    print(max(r))
    
    plt.scatter(X,Y)
    plt.plot(X, y_poly_pred, color='m')
    save_to_csv(X, y_poly_pred) #cieka 
    plt.show()

def poli_regression(Xin, Yin):
    first_degree = 2
    last_degree = 22
    degree_step = 1
    r= []

    plt.scatter(Xin, Yin)
    
    for degree in range(first_degree, last_degree, degree_step):
        model = make_pipeline(PolynomialFeatures(degree), Ridge())
        model.fit(Xin.reshape(-1, 1), Yin)
        Yout = model.predict(Xin.reshape(-1, 1))
        r2 = r2_score(Yin,Yout)
        r.append((r2,degree))
        
    #print(max(r))
    najlepszy = max(r)
    #print(najlepszy[1])
    model = make_pipeline(PolynomialFeatures(najlepszy[1]), Ridge())
    model.fit(Xin.reshape(-1, 1), Yin)
    Yout = model.predict(Xin.reshape(-1, 1))
    plt.plot(Xin, Yout,label="degree %d" % najlepszy[1], color ='red')
    save_to_csv(Xin, Yout) #cieka 

    plt.legend(loc='lower left')
    plt.title('Polynomial regression')
    plt.show()
